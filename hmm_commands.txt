# download hmm model from pfam db
wget http://pfam.xfam.org/family/PF00931/hmm -O PF00931_NB_ARC

# identify protein families
hmmsearch --tblout hua_s1.table -o hua_s1.result PF00931_NB_ARC hua_s1.faa

# filter
cat hua_s1.table |  awk '$5<1e-04 {print}' | grep -v "#" > filtered01.txt
